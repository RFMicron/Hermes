142 void gen2Select(struct gen2SelectParams *p)
143 {
144 #if !RUN_ON_AS3980 // no select on AS3980 available
145  u16 len = p->mask_len;
146  u8 ebvLen;
147  u16 rxbits = 1;
148  u8 *mask = p->mask;
149  u8 j,i;
150  u8 resp_null;
151  u8 jLeftBits;
152  u8 posTruncbit;
153  as3993ClrResponse();
154  memset(buf_,0,sizeof(buf_));
155 
156  buf_[0] = ((EPC_SELECT<<4)&0xF0) | ((p->target<<1)&0x0E) | ((p->action>>2)&0x01);
157  buf_[1] = ((p->action<<6)&0xC0) | ((p->mem_bank<<4)&0x30);
158  ebvLen = gen2InsertEBV(p->mask_address, buf_+1, 4);
159  buf_[1+ebvLen] = ((p->mask_len>>4)&0x0F);
160  buf_[2+ebvLen] = ((p->mask_len<<4)&0xF0);
161 
162  i = 2+ebvLen; /* the counter set to last buffer location */
163  for (j = len; j >= 8 ; j -= 8, mask++)
164  {
165  buf_[i] |= 0x0F & (*mask >> 4);
166  i++;
167  buf_[i] = 0xF0 & (*mask << 4);
168  }
169 
170  if (j == 0)
171  {
172  buf_[i] |= ((p->truncation<<3)&0x08)/*Truncate*/;
173  }
174  else // if length is not dividable by 8
175  { jLeftBits = 0xFF << (8-j);
176 
177  buf_[i] |= (jLeftBits >> 4) & (*mask >> 4);
178  buf_[i+1] = (jLeftBits << 4) & (*mask << 4);
179  // add truncation bit
180  posTruncbit = (p->truncation & 0x01) <<(7-j);
181  buf_[i] |= (posTruncbit >> 4) ;
182  buf_[i+1] |= (posTruncbit << 4) ;
183  }
184 
185  len += 21 + ebvLen * 8; /* Add the bits for EPC command before continuing */
186 
187  /* Pseudo 1-bit receival with small timeout to have AS3993 state machine
188  finished and avoiding spurious interrupts (no response) */
189  as3993TxRxGen2Bytes(AS3993_CMD_TRANSMCRC, buf_, len, &resp_null, &rxbits, 1, 0, 1);
190 #endif
191 }

244 /*
245  * Sends a Query, QueryRep or QueryAdjust command and checks for replies from tags.
246  * If a tag is found, the Ack command is sent. If fast mode is not enabled the Reg_RN
247  * command will be sent after receiving the Ack reply.\n
248  * All of the data which is retrieved from the tag is written into parameter tag.
249  * \param tag The reply data of the tag is written into this data structure
250  * \param qCommand The command which should be sent. Should be Query, QueryRep or QueryAdjust
251  * \param q Q as specified in the Gen2 spec
252  * \param fast If fast mode is activated no Req_RN command will be sent after receiving the
253  * Ack reply. If no QueryRep, QueryAdjust or Reg_RN is sent to the tag within T2 the tag will
254  * go back to arbitrate state.
255  * \param followCommand The command which will be sent after a slot has been processed
256  * successfully (valid tag response). Usually this will be used to send a QueryRep
257  * immediately after receiving a tag response to trigger a change of the current
258  * session flag.
259  * \return 1 if one tag has been succesfully read, 0 if no response in slot, -1 if error occured (collision)
260  */
261 static s8 gen2Slot(Tag *tag, u8 qCommand, u8 q, BOOL fast, u8 followCommand)
262 {
263  u16 rxlen;
264  s8 ret = 0;
265 
266  /*********************************************************************************/
267  /* 1. Send proper query command */
268  /*********************************************************************************/
269  rxlen = 16;
270  switch (qCommand)
271  {
272  case AS3993_CMD_QUERY:
273  gen2PrepareQueryCmd(buf_, q);
274  ret = as3993TxRxGen2Bytes(AS3993_CMD_QUERY,buf_,16,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
275  break;
276  case 0: //query rep has already been sent
277  ret = as3993TxRxGen2Bytes(0,NULL,0,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
278  break;
279  default:
280  ret = as3993TxRxGen2Bytes(qCommand,NULL,0,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
281  }
282  if (ret == AS3993_ERR_NORES)
283  {
284  EPCLOG(" query -> noresp\n");
285  return 0;
286  }
287  if (ret < 0)
288  {
289  EPCLOG(" query -> err %hhx\n", ret);
290  return -1;
291  }
292 
293  /*********************************************************************************/
294  /* 2. Sent ACK, Receive pc, epc, send REQRN immediately after FIFO has the data */
295  /*********************************************************************************/
296  rxlen = sizeof(buf_)*8; //receive all data, length auto-calculated by AS3993
297  ret = as3993TxRxGen2Bytes(0,buf_,0,buf_,&rxlen,gen2IntConfig.no_resp_time,fast?followCommand:AS3993_CMD_REQRN,0);
298  tag->pc[0] = buf_[0];
299  tag->pc[1] = buf_[1];
300  if (ret < 0 || rxlen < 16)
301  {
302  EPCLOG(" ack rx rest(pc=%hhx) -> err %hhx\n", tag->pc[0], ret);
303  return -1;
304  }
305 #if RUN_ON_AS3980
306  tag->epclen = (rxlen+7)/8-4; //on AS3980 crc is in the fifo, omit it.
307 #else
308  tag->epclen = (rxlen+7)/8-2;
309 #endif
310  if(tag->epclen > EPCLENGTH)
311  tag->epclen = EPCLENGTH;
312  memcpy(tag->epc, buf_+2, tag->epclen);
313 
314  if (fast)
315  {
316  gen2GetAgcRssi(&tag->agc, &tag->rssi);
317  if (((tag->pc[0] & 0xF8) >> 2) != tag->epclen)
318  {
319  return -1;
320  }
321  return 1;
322  }
323  /*********************************************************************************/
324  /* 3. Receive the two bytes handle */
325  /*********************************************************************************/
326  rxlen = 32;
327  ret = as3993TxRxGen2Bytes(0,NULL,0,tag->handle,&rxlen,gen2IntConfig.no_resp_time,followCommand,0);
328  gen2GetAgcRssi(&tag->agc, &tag->rssi);
329  if (ret < 0)
330  {
331  EPCLOG(" reqrn -> err %hhx\n", ret);
332  return -1;
333  }
334  if (((tag->pc[0] & 0xF8) >> 2) != tag->epclen)
335  {
336  return -1;
337  }
338  return 1;
339 }

437 s8 gen2AccessTag(Tag const * tag, u8 const * password)
438 {
439  s8 ret;
440  s8 error;
441  u8 count;
442  u16 rxcount;
443  u8 tagResponse[5];
444  u8 temp_rn16[2];
445 
446  for (count = 0; count < 2; count++)
447  {
448  error = gen2ReqRNHandleChar(tag->handle, temp_rn16);
449  if (error)
450  {
451  return(error);
452  }
453 
454  buf_[0] = EPC_ACCESS;
455  buf_[1] = password[0] ^ temp_rn16[0];
456  buf_[2] = password[1] ^ temp_rn16[1];
457  buf_[3] = tag->handle[0];
458  buf_[4] = tag->handle[1];
459 
460  rxcount = 32;
461  ret = as3993TxRxGen2Bytes(AS3993_CMD_TRANSMCRC,buf_,40,tagResponse,&rxcount,gen2IntConfig.no_resp_time,0,1);
462  if (ret < 0)
463  {
464  EPCLOG("access failed, tx-ret: %hhx\n", ret);
465  return GEN2_ERR_ACCESS;
466  }
467  password += 2; /* Increase pointer by two to fetch the next bytes;- */
468 
469  if ((tagResponse[1] != tag->handle[1]) ||
470  (tagResponse[0] != tag->handle[0]))
471  {
472  EPCLOG("handle not correct\n");
473  return GEN2_ERR_ACCESS;
474  }
475 #if EPCDEBUG
476  if (count ==0) EPCLOG("first part of access ok\n");
477  if (count ==1) EPCLOG("second part of access ok\n");
478 #endif
479  }
480  return ret;
481 }

518 s8 gen2KillTag(Tag const * tag, u8 const * password, u8 rfu, u8 recom, u8* tag_error)
519 {
520  s8 error;
521  u8 count;
522  u8 temp_rn16[2];
523  u16 rxbits = 32;
524  u8 cmd = AS3993_CMD_TRANSMCRC;/* first command has no header Bit */
525  u8 no_resp_time = gen2IntConfig.no_resp_time;
526 
527  *tag_error = 0xa5;
528  for (count = 0; count < 2; count++)
529  {
530  error = gen2ReqRNHandleChar(tag->handle, temp_rn16);
531  if (error)
532  {
533  break;
534  }
535 
536  if (count==1)
537  { /* Values for second part of kill */
538  cmd = AS3993_CMD_TRANSMCRCEHEAD; /* expect header bit */
539  rxbits = 32 + 1; /* add header bit */
540  rfu = recom; /* different data for rfu/recom */
541  no_resp_time = 0xff; /* waiting time up to 20ms */
542  }
543 
544  buf_[0] = EPC_KILL;
545 
546  buf_[1] = password[0] ^ temp_rn16[0];
547  buf_[2] = password[1] ^ temp_rn16[1];
548 
549  buf_[3] = ((rfu << 5) & 0xE0);
550  insertBitStream(&buf_[3], tag->handle, 2, 5);
551 
552  error = as3993TxRxGen2Bytes(cmd, buf_, 43, buf_, &rxbits, no_resp_time, 0, 1);
553 
554  if ( error )
555  {
556  break;
557  }
558  password += 2;
559  }
560  if(ERR_CHIP_HEADER == error && rxbits) *tag_error = buf_[0];
561  return error;
562 }

678 unsigned gen2SearchForTags(Tag *tags_
679  , u8 maxtags
680  , u8 q
681  , BOOL (*cbContinueScanning)(void)
682  , BOOL singulate
683  , BOOL toggleSession
684  , s8 (*followTagCommand)(Tag *tag)
685  )
686 {
687  u16 num_of_tags = 0;
688  u16 collisions = 0;
689  u16 slot_count;
690  u8 i = 0;
691  u8 addRounds = 1; /* the maximal number of rounds performed */
692  u8 cmd = AS3993_CMD_QUERY;
693  u8 followCmd = 0;
694 
695 #if !RUN_ON_AS3980
696  if (toggleSession)
697  followCmd = AS3993_CMD_QUERYREP;
698 #endif
699 
700  if (followTagCommand != NULL)
701  {
702  followCmd = 0; //no follow Command
703  singulate = 1; // not fast
704  }
705  as3993AntennaPower(1);
706  as3993ContinuousRead(AS3993_REG_IRQSTATUS1, 2, &buf_[0]); // ensure that IRQ bits are reset
707  as3993ClrResponse();
708 
709  EPCLOG("Searching for Tags, maxtags=%hhd, q=%hhd\n",maxtags,q);
710  EPCLOG("-------------------------------\n");
711 
712  for (i=0; i < maxtags; i++) /*Reseting the TAGLIST */
713  {
714  tags_[i].rn16[0] = 0;
715  tags_[i].rn16[1] = 0;
716  tags_[i].epclen=0;
717  }
718  do
719  {
720  BOOL goOn;
721  collisions = 0;
722  slot_count = 1UL<<q; /*get the maximum slot_count */
723  do
724  {
725  if (num_of_tags >= maxtags)
726  {/* ERROR it is not possible to store more than maxtags Tags */
727  break;
728  }
729  EPCLOG("next slot, command: %x\n", cmd);
730  slot_count--;
731 
732  switch (gen2Slot(tags_+num_of_tags, cmd, q, !singulate, followCmd))
733  {
734  case -1:
735  //EPCLOG("collision\n");
736  collisions++;
737  cmd = AS3993_CMD_QUERYREP;
738  break;
739  case 1:
740  if (followCmd)
741  cmd = 0; // query_rep has already been sent as followCmd
742  else
743  cmd = AS3993_CMD_QUERYREP;
744  if (followTagCommand != NULL)
745  {
746  followTagCommand(tags_+num_of_tags);
747  cmd = AS3993_CMD_QUERYREP;
748  }
749  num_of_tags++;
750  break;
751  case 0:
752  //EPCLOG("NO EPC response -> empty Slot\n");
753  cmd = AS3993_CMD_QUERYREP;
754  break;
755  default:
756  break;
757  }
758  goOn = cbContinueScanning();
759  } while (slot_count && goOn );
760  addRounds--;
761  EPCLOG("q=%hhx, collisions=%x, num_of_tags=%x",q,collisions,num_of_tags);
762  if( collisions )
763  if( collisions >= (1UL<<q) /4)
764  {
765  q++;
766  EPCLOG("->++\n");
767  cmd = AS3993_CMD_QUERYADJUSTUP;
768  }
769  else if( collisions < (1UL<<q) /8)
770  {
771  q--;
772  EPCLOG("->--\n");
773  cmd = AS3993_CMD_QUERYADJUSTDOWN;
774  }
775  else
776  {
777  EPCLOG("->==\n");
778  cmd = AS3993_CMD_QUERYADJUSTNIC;
779  }
780  else
781  {
782  EPCLOG("->!!\n");
783  addRounds = 0;
784  }
785  }while(num_of_tags < maxtags && addRounds && cbContinueScanning() );
786 
787  // If a tag is found in the last slot of an inventory round, follow Command QueryRep will be executed.
788  // Add some time to send this command to the tag over the field.
789  // If function will be left to early and field will be switched off the tag will not invert the inventoried flag.
790  // Problem especially for S2 and S3.
791  // If there was a follow Tag Command a Query Rep. has to be sent to invert the session flag.
792 
793  if (followTagCommand == NULL)
794  {
795  delay_us(150);
796  }
797  else
798  {
799  as3993SingleCommand(AS3993_CMD_QUERYREP);
800  as3993WaitForResponse(RESP_TXIRQ);
801  as3993ClrResponse();
802  }
803 
804 #if EPCDEBUG
805  EPCLOG("-------------------------------\n");
806  EPCLOG("%hx Tags found", num_of_tags);
807  EPCLOG("\n");
808 #endif
809  return num_of_tags;
810 }

930 void gen2Configure(const struct gen2Config *config)
931 {
932  /* depending on link frequency setting adjust */
933  /* registers 01, 02, 03, 04, 05, 06, 07, 08 and 09 */
934  u8 reg[9];
935  u8 session = config->session;
936  gen2IntConfig.DR = 1;
937  gen2IntConfig.config = *config;
938  if (session > GEN2_IINV_S3)
939  session = GEN2_IINV_S0; /* limit SL and invalid settings */
940  if (gen2IntConfig.config.miller == GEN2_COD_FM0 || gen2IntConfig.config.miller == GEN2_COD_MILLER2)
941  gen2IntConfig.config.trext = TREXT_ON;
942 
943  switch (config->linkFreq) {
944  case GEN2_LF_640: /* 640kHz */
945  reg[0] = 0x20; /* AS3993_REG_TXOPTIONS */
946  reg[1] = 0xF0; /* AS3993_REG_RXOPTIONS */
947  reg[2] = 0x01; /* AS3993_REG_TRCALHIGH */
948  reg[3] = 0x4D; /* AS3993_REG_TRCALLOW */
949  reg[4] = 0x03; /* AS3993_REG_AUTOACKTIMER */
950  reg[5] = 0x02; /* AS3993_REG_RXNORESPONSEWAITTIME */
951  reg[6] = 0x01; /* AS3993_REG_RXWAITTIME */
952  if (gen2IntConfig.config.miller > GEN2_COD_MILLER2)
953  reg[7] = 0x04; /* AS3993_REG_RXFILTER */
954  else
955  reg[7] = 0x07; /* AS3993_REG_RXFILTER */
956  break;
957  case GEN2_LF_320: /* 320kHz */
958  reg[0] = 0x20;
959  reg[1] = 0xC0;
960  if (gen2IntConfig.config.tari == TARI_625) /* if Tari = 6.25us */
961  { /* TRcal = 25us */
962  gen2IntConfig.DR = 0;
963  reg[2] = 0x00;
964  reg[3] = 0xFA;
965  } else
966  { /* TRcal = 66.7us */
967  reg[2] = 0x02;
968  reg[3] = 0x9B;
969  }
970  reg[4] = 0x04;
971  reg[5] = 0x03;
972  reg[6] = 0x02;
973  if (gen2IntConfig.config.miller > GEN2_COD_MILLER2)
974  reg[7] = 0x24; /* AS3993_REG_RXFILTER */
975  else
976  reg[7] = 0x27; /* AS3993_REG_RXFILTER */
977  break;
978  case GEN2_LF_256: /* 256kHz */
979  reg[0] = 0x20;
980  reg[1] = 0x90;
981  if (gen2IntConfig.config.tari == TARI_625) /* Tari = 6.25us */
982  { /* TRcal = 31.3us */
983  gen2IntConfig.DR = 0;
984  reg[2] = 0x01;
985  reg[3] = 0x39;
986  }
987  else /* Tari = 25us or 12.5us */
988  { /* TRcal = 83.3us */
989  reg[2] = 0x03;
990  reg[3] = 0x41;
991  }
992  reg[4] = 0x05;
993  reg[5] = 0x05;
994  reg[6] = 0x04;
995  if (gen2IntConfig.config.miller > GEN2_COD_MILLER2)
996  reg[7] = 0x34; /* AS3993_REG_RXFILTER */
997  else
998  reg[7] = 0x37; /* AS3993_REG_RXFILTER */
999  break;
1000  case GEN2_LF_213: /* 213.3kHz */
1001  reg[0] = 0x20;
1002  reg[1] = 0x80;
1003  if (gen2IntConfig.config.tari == TARI_625) /* Tari = 6.25us */
1004  { /* TRcal = 37.5us */
1005  gen2IntConfig.DR = 0;
1006  reg[2] = 0x01;
1007  reg[3] = 0x77;
1008  }
1009  else
1010  { /* TRcal = 100us */
1011  reg[2] = 0x03;
1012  reg[3] = 0xE8;
1013  }
1014  reg[4] = 0x06;
1015  reg[5] = 0x05;
1016  reg[6] = 0x05;
1017  if (gen2IntConfig.config.miller > GEN2_COD_MILLER2)
1018  reg[7] = 0x34; /* AS3993_REG_RXFILTER */
1019  else
1020  reg[7] = 0x37; /* AS3993_REG_RXFILTER */
1021  break;
1022  case GEN2_LF_160: /* 160kHz */
1023  reg[0] = 0x20;
1024  reg[1] = 0x60;
1025  if (gen2IntConfig.config.tari == TARI_125) /* Tari = 12.5us */
1026  { /* TRcal = 50us */
1027  gen2IntConfig.DR = 0;
1028  reg[2] = 0x01;
1029  reg[3] = 0xF4;
1030  }
1031  else
1032  { /* TRcal = 1333.3us */
1033  reg[2] = 0x05;
1034  reg[3] = 0x35;
1035  }
1036  reg[4] = 0x0A;
1037  reg[5] = 0x05;
1038  reg[6] = 0x08;
1039  if (gen2IntConfig.config.miller > GEN2_COD_FM0)
1040  reg[7] = 0x3F; /* AS3993_REG_RXFILTER */
1041  else
1042  reg[7] = 0xBF; /* AS3993_REG_RXFILTER */
1043  break;
1044  case GEN2_LF_40: /* 40kHz */
1045  reg[0] = 0x30; /* AS3993_REG_TXOPTIONS */
1046  reg[1] = 0x00; /* AS3993_REG_RXOPTIONS */
1047  reg[2] = 0x07; /* AS3993_REG_TRCALHIGH */
1048  reg[3] = 0xD0; /* AS3993_REG_TRCALLOW */
1049  reg[4] = 0x3F; /* AS3993_REG_AUTOACKTIMER */
1050  reg[5] = 0x0C; /* AS3993_REG_RXNORESPONSEWAITTIME */
1051  reg[6] = 0x24; /* AS3993_REG_RXWAITTIME */
1052  reg[7] = 0xFF; /* AS3993_REG_RXFILTER */
1053  gen2IntConfig.DR = 0;
1054  break;
1055  default:
1056  return; /* use preset settings */
1057  }
1058  reg[0] |= gen2IntConfig.config.tari;
1059  reg[1] = (reg[1] & ~0x0F)
1060  | gen2IntConfig.config.miller
1061  | (gen2IntConfig.config.trext<<3);
1062 
1063  gen2IntConfig.no_resp_time = reg[5];
1064  /* Modify only the gen2 relevant settings */
1065  as3993ContinuousWrite(AS3993_REG_TXOPTIONS, reg+0, 8);
1066 
1067  reg[0] = as3993SingleRead(AS3993_REG_TXSETTING);
1068  reg[0] = (reg[0] & ~0x03) | gen2IntConfig.config.session;
1069  as3993SingleWrite(AS3993_REG_TXSETTING, reg[0]);
1070  reg[0] = as3993SingleRead(AS3993_REG_PROTOCOLCTRL);
1071  reg[0] = reg[0] & ~0x07;
1072  as3993SingleWrite(AS3993_REG_PROTOCOLCTRL, reg[0]);
1073 }
