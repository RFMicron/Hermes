605 static void as3993LockPLL(void)
606 {
607  u8 buf;
608  u8 var;
609  u16 i;
610 
611  u8 vco_voltage;
612  buf = as3993SingleRead(AS3993_REG_STATUSPAGE);
613  buf &= ~0x30;
614  buf |= 0x10; /* have vco_ri in aglstatus */
615  as3993SingleWrite(AS3993_REG_STATUSPAGE, buf);
616 
617  buf = as3993SingleRead(AS3993_REG_VCOCONTROL);
618  buf |= 0x80; /* set mvco bit */
619  as3993SingleWrite(AS3993_REG_VCOCONTROL, buf);
620  delay_ms(1); /* give PLL some settling time, should be around 500us */
621 
622  vco_voltage = as3993SingleRead(AS3993_REG_AGL) & 0x07;
623 
624  buf &= ~0x80; /* reset mvco bit */
625  as3993SingleWrite(AS3993_REG_VCOCONTROL, buf);
626 
627  if ( vco_voltage <= 1 || vco_voltage >= 6 )
628  {
629  i=0;
630  do
631  {
632  i++;
633  as3993SingleCommand(AS3993_CMD_VCO_AUTO_RANGE);
634  delay_ms(10); /* Please keep in mind, that the Auto Bit procedure will take app. 6 ms whereby the locktime of PLL is just 400us */
635  var=as3993SingleRead(AS3993_REG_AGCANDSTATUS);
636  } while ( (var & 0x02)==0 && (i<3));/* wait for PLL to be locked and give a few attempts */
637  }
638 }

641 void as3993SetBaseFrequency(u8 regs, u32 frequency)
642 {
643  u8 buf[3];
644  u8 statusreg;
645  u16 ref=0, i, j, x, reg_A,reg_B;
646  u32 divisor;
647 
648  statusreg = as3993SingleRead(AS3993_REG_STATUSCTRL);
649  as3993SingleWrite(AS3993_REG_STATUSCTRL, statusreg & 0xfe);
650  if (regs == AS3993_REG_PLLMAIN1)
651  {
652  as3993ContinuousRead(AS3993_REG_PLLMAIN1, 3, buf);
653  switch (buf[0]& 0x70)
654  {
655 
656  case 0x40: {
657  ref=125;
658  } break;
659  case 0x50: {
660  ref=100;
661  } break;
662  case 0x60: {
663  ref=50;
664  } break;
665  case 0x70: {
666  ref=25;
667  } break;
668  default: {
669  ref=0;
670  }
671  }
672  }
673 
674  if (ref == 0)
675  {
676  return;
677  }
678

934 s8 as3993GetSensitivity( )
935 {

938

940  reg0a = as3993SingleRead(AS3993_REG_RXMIXERGAIN);

962  {

964  }

968  { /* RX Gain direction: increase */
969  sensitivity -= gain;
970  }
