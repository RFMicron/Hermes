82 static void systemClockFindPllPrescaler ( u32 frequency )
83 {
84  u32 border = PLL_OUT_NOMINAL / 2 ;
85  u16 divider;
86  if ( frequency >= PLL_OUT_NOMINAL )
87  {
88  frequency = PLL_OUT_NOMINAL;
89  }
90  for ( divider = 0; divider < 3; divider++ )
91  {
92  u32 offset = border / 2;
93  if ( frequency > border + offset )
94  {
95  break; /* found it */
96  }
97  border /= 2;
98  }
99  /* either found it or stopped by 3 (which is smallest possible divider) */
100  CLKDIVbits.CPDIV = divider;
101 #ifndef __PIC24FJ256GB106__
102  CLKDIVbits.PLLEN = 1; /* enable pll */
103 #endif
104  /* wait for lock of pll */
105  while ( ! OSCCONbits.LOCK );
106 }
107 #endif

149 u32 systemClockGetFcy ( )
150 {
151  u32 frequency = 0;
152  switch ( OSCCONbits.COSC )
153  {
154 
155  case CLK_SRC_LPRC:
156  frequency = LPRC_NOMINAL;
157  break;
158 
159  case CLK_SRC_SOSC:
160  frequency = SOSC_NOMINAL;
161  break;
162 
163  case CLK_SRC_POSC_PLL:
164  if ( POSC_NOMINAL )
165  {
166  frequency = systemClockFromPllPrescaler( );
167  }
168  break;
169 
170  case CLK_SRC_POSC:
171  frequency = POSC_NOMINAL;
172  break;
173 
174  case CLK_SRC_FRCDIV:
175  frequency = systemClockFromFrcPostscaler( );
176  break;
177 
178  case CLK_SRC_FRCPLL:
179  frequency = systemClockFromPllPrescaler( );
180  break;
181 
182  case CLK_SRC_FRC:
183  frequency = FRC_NOMINAL;
184  break;
185  }
186  theSystemClock = frequency / 2; /* the theSystemClock is always half the system clock frequency */
187  return theSystemClock;
188 }
