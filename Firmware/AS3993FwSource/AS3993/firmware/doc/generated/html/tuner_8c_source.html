166 void tunerSetCap(const TunerConfiguration *config, u8 component, u8 val )
167 {
168  spiConfig_t tunerConfig, oldConfig;
169 
170  if (tunerCheckConfig(config, component))
171  return; //we have an invalid component.
172 
173  /* reconfigure spi to 2MHz */
174  tunerConfig.frequency = 2000000ULL;
175  tunerConfig.instance = SPI1;
176  tunerConfig.clockPhase = 0;
177  tunerConfig.clockPolarity = 0;
178  tunerConfig.deviceId = 0;
179  spiInitialize(&tunerConfig, &oldConfig);
180 
181  switch (component)
182  {
183  case TUNER_CIN:
184  config->cinSenFn(1);
185  break;
186  case TUNER_CLEN:
187  config->clenSenFn(1);
188  break;
189  case TUNER_COUT:
190  config->coutSenFn(1);
191  break;
192  }
193 
194  spiTxRx(&val, 0, 1);
195 
196  switch (component)
197  {
198  case TUNER_CIN:
199  config->cinSenFn(0);
200  break;
201  case TUNER_CLEN:
202  config->clenSenFn(0);
203  break;
204  case TUNER_COUT:
205  config->coutSenFn(0);
206  break;
207  }
208  /* restore spi to default */
209  spiInitialize(&oldConfig, 0);
210 }
211 u16 tunerGetReflected(void)
212 {
213  u16 r; u32 refl;
214  s8 i, q;
215 
216  delay_us(300);
217  r = as3993GetReflectedPower();
218  i = r & 0xff;
219  q = r >> 8;
220  refl = (i * i) + (q * q);
221 
222  r = refl;
223  if (refl > 0xffff) r= 0xffff;
224 
225  return r;

228 u8 tunerClimbOneParam(const TunerConfiguration *config, u8 el, u8 *elval, u16 *reflectedPower, u16 *maxSteps)
229 {
230  u16 refl, start = *reflectedPower;
231  s8 dir = 0;
232  s8 add = 0;
233  u8 improvement = 3;
234  u8 bestelval;
235 
236  if (tunerCheckConfig(config, el))
237  return 0; //we have an invalid component.
238 
239  if (*maxSteps == 0)
240  {
241  return 0;
242  }
243 
244  TLOG("start: el %hhx val %hhx r %hx\n",el,*elval,*reflectedPower);
245  if ( *elval != 0 )
246  {
247  tunerSetCap(config, el, *elval - 1 );
248  refl = tunerGetReflected();
249  TLOG("-1: r %hx\n",refl);
250  if ( refl <= *reflectedPower )
251  {
252  *reflectedPower = refl;
253  dir = -1;
254  }
255  }
256 
257  if ( *elval < 31 )
258  {
259  tunerSetCap(config, el, *elval + 1 );
260  refl = tunerGetReflected();
261  TLOG("+1: r %hx\n",refl);
262  if ( refl <= *reflectedPower )
263  {
264  *reflectedPower = refl;
265  dir = 1;
266  }
267  }
268 
269  /* if it's possible we try to change the value by 2 and check what direction is better.
270  * This improves tuning in the case of a local minima. */
271  if ( *elval > 1 )
272  {
273  tunerSetCap(config, el, *elval - 2 );
274  refl = tunerGetReflected();
275  TLOG("-2: r %hx\n",refl);
276  if ( refl <= *reflectedPower )
277  {
278  *reflectedPower = refl;
279  dir = -1;
280  add = -1;
281  }
282  }
283 
284  if ( *elval < 30 )
285  {
286  tunerSetCap(config, el, *elval + 2 );
287  refl = tunerGetReflected();
288  TLOG("+2: r %hx\n",refl);
289  if ( refl <= *reflectedPower )
290  {
291  *reflectedPower = refl;
292  dir = 1;
293  add = 1;
294  }
295  }
296 
297  *elval += add;
298  *elval += dir;
299  bestelval = *elval;
300  TLOG("decision: dir %hhx add %hhx\n",dir, add);
301 
302  if (dir!=0)
303  {
304  (*maxSteps)--;
305  while ( improvement && *maxSteps )
306  {
307  if ( *elval == 0 ) break;
308  if ( *elval == 31 ) break;
309  tunerSetCap(config, el, *elval + dir );
310  refl = tunerGetReflected();
311  TLOG("t %hhx: r %hx\n",*elval + dir, refl);
312  if ( refl <= *reflectedPower )
313  {
314  (*maxSteps)--;
315  *reflectedPower = refl;
316  *elval += dir;
317  bestelval = *elval;
318  improvement = 3; /* we don't want to stop when a local minima is hit, so we
319  continue tuning for 3 more steps, even if they are worse than the one before.
320  If it does not improve in this 3 steps, we will go back to the best value. */
321  }else{
322  improvement--;
323  }
324  }
325  }
326  *elval = bestelval;
327  tunerSetCap(config, el, *elval);
328  TLOG("result val %hhx: r %hx\n",*elval, *reflectedPower);
329  return start > (*reflectedPower);
330 }

332 void tunerOneHillClimb(const TunerConfiguration *config, TunerParameters *p, u16 maxSteps)
333 {
334  u8 improvement = 1;
335 
336  tunerSetTuning(config, p->cin, p->clen, p->cout);
337 
338  p->reflectedPower = tunerGetReflected();
339 
340  while (maxSteps && improvement)
341  {
342  improvement = 0;
343  if (config->cinSenFn)
344  improvement |= tunerClimbOneParam(config, TUNER_CIN, &p->cin, &p->reflectedPower, &maxSteps);
345  if (config->clenSenFn)
346  improvement |= tunerClimbOneParam(config, TUNER_CLEN, &p->clen, &p->reflectedPower, &maxSteps);
347  if (config->coutSenFn)
348  improvement |= tunerClimbOneParam(config, TUNER_COUT, &p->cout, &p->reflectedPower, &maxSteps);
349  }
350 
351  tunerSetTuning(config, p->cin, p->clen, p->cout);
352 }
