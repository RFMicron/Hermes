105 /*************************************************************************
106  Function:
107  BOOL USBSleepOnSuspend(void)
108  Summary:
109  Places the PIC24F devices in sleep after enabling the USB activity flag to
110  wake it back up.
111  Conditions:
112  IPL (in the SR register) must be non-zero.
113  Input:
114  None
115  Return:
116  TRUE - if entered sleep successfully
117  FALSE - if there was an error entering sleep
118  Remarks:
119  None
120  *************************************************************************/
121 BOOL USBSleepOnSuspend(void)
122 {
123  unsigned int U1EIE_save, U1IE_save, U1OTGIE_save;
124  unsigned char USB1IE_save;
125 
126  //Save the old interrupt and CPU settings
127  U1EIE_save = U1EIE;
128  U1IE_save = U1IE;
129  U1OTGIE_save = U1OTGIE;
130  USB1IE_save = IEC5bits.USB1IE;
131 
132 #if defined(USB_POLLING)
133  //If IPL is equal to 0 then there is no way for the USB module to
134  // generate an interrupt to wake up the device.
135  if(_IPL == 0)
136  {
137  return FALSE;
138  }
139 #endif
140 
141  //Disable all USB interrupts
142  // 2011/12/13, mar: moved this code down where we do not exit in
143  // polling mode. In Polling all usb-ir flags were disabled, and
144  // we could never change to the activated state as the activate
145  // enable bit was cleared. If we do not use any interrupts, the
146  // IPL level is 0.
147  U1EIE = 0;
148  U1IE = 0;
149  U1OTGIE = 0;
150 
151 #if defined(USB_POLLING)
152  //Set the interrupt priority to a level that will wake up the part (>0)
153  // but will not cause a interrupt vector jump (USB1IP<=IPL)
154  _USB1IP = 1;
155 #endif
156 
157 
158 
159  //Enable the interrupt
160  IFS5bits.USB1IF = 0;
161  U1OTGIEbits.ACTVIE = 1;
162  USBClearInterruptFlag(USBActivityIFReg,USBActivityIFBitNum);
163  IEC5bits.USB1IE = 1;
164 
165  {
166  /* PIC24FJ64GB004 Family Silicon Errata: point 7 */
167  unsigned char spienbits[2];
168 #ifdef __PIC24FJ128GB202__
169  spienbits[0] = SPI1CON1Lbits.SPIEN;
170  if(SPI2CON2L&0x8000)
171  spienbits[1]=1;
172  else
173  spienbits[1]=0;
174 
175  SPI1CON1Lbits.SPIEN = 0;
176  SPI2CON1L = (SPI2CON1L&0x7FFF);
177 
178  Sleep();
179 
180  SPI1CON1Lbits.SPIEN = spienbits[0];
181  SPI2CON1L = ((SPI2CON1L&0x7FFF) | (spienbits[1]<<15));
182 #else
183  spienbits[0] = SPI1STATbits.SPIEN;
184  spienbits[1] = SPI2STATbits.SPIEN;
185 
186  SPI1STATbits.SPIEN = 0;
187  SPI2STATbits.SPIEN = 0;
188 
189  Sleep();
190 
191  SPI1STATbits.SPIEN = spienbits[0];
192  SPI2STATbits.SPIEN = spienbits[1];
193 #endif
194  }
195 
196 #if defined(USB_POLLING)
197  //Disable the interrupt
198  _USB1IP = 0;
199 #endif
200 
201  //restore the previous interrupt settings
202  IEC5bits.USB1IE = USB1IE_save;
203  U1EIE = U1EIE_save;
204  U1IE = U1IE_save;
205  U1OTGIE = U1OTGIE_save;
206 
207  return TRUE;
208 }
